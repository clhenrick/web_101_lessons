Timing Functions

  // setTimeout
  // JS will wait x milliseconds before executing the code.
  
  window.setTimeout(function(){
    alert("hello!");
  }, 2000); 
  // the above code will run 2 seconds after the window loads.
  

  // setInterval
  // JS will wait x milliseconds between repititions
  
  window.setInterval(function(){
    console.log("hello!");
  }, 2000);
  // the above code will continue for infinity every 2 seconds.
